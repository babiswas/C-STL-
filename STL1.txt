#include<iostream>
#include<vector>
using namespace std;

int main()
{
    vector<int> v;
    v.push_back(10);
    v.push_back(20);
    v.push_back(30);
    v.push_back(40);
    vector<int>::iterator itr=v.begin();
    cout<<*itr<<endl;
    cout<<"Size="<<v.size()<<endl;
    for(int i=0;i<v.size();i++)
    {
        cout<<*(itr+i)<<endl;
    }
    v.clear();
    cout<<"Size="<<v.size()<<endl;
    return 0;
}
